USE master;
GO

IF EXISTS (SELECT name FROM sys.databases WHERE name = N'InventarioEquipos')

    ALTER DATABASE InventarioEquipos SET SINGLE_USER WITH ROLLBACK IMMEDIATE;
    DROP DATABASE InventarioEquipos;

CREATE DATABASE InventarioEquipos;
GO

USE InventarioEquipos;
GO

-- ============================================================
-- TABLAS INDEPENDIENTES
-- ============================================================

CREATE TABLE dbo.SISTEMA_OPERATIVO (
    Id_Sistema_Operativo  INT            NOT NULL IDENTITY(1,1),
    Sistema_Operativo     NVARCHAR(100)  NOT NULL,
    Sistema_Licencia      NVARCHAR(100)  NOT NULL,
    CONSTRAINT PK_SistemaOperativo PRIMARY KEY (Id_Sistema_Operativo)
);
GO

CREATE TABLE dbo.ALMACEN (
    Id_Codigo_Almacen  INT            NOT NULL IDENTITY(1,1),
    Dependencia        NVARCHAR(150)  NOT NULL,
    Fecha              DATE           NOT NULL,
    Codigo_Tecnologia  NVARCHAR(50)   NOT NULL,
    Usuario            NVARCHAR(100)  NOT NULL,
    CONSTRAINT PK_Almacen PRIMARY KEY (Id_Codigo_Almacen)
);
GO

CREATE TABLE dbo.DISCO_DURO (
    Id_DiscoDuro        INT            NOT NULL IDENTITY(1,1),
    DiscoDuro_Capacidad NVARCHAR(50)   NOT NULL,
    DiscoDuro_Marca     NVARCHAR(100)  NOT NULL,
    CONSTRAINT PK_DiscoDuro PRIMARY KEY (Id_DiscoDuro)
);
GO

CREATE TABLE dbo.PROCESADOR (
    Codigo_Procesador    INT           NOT NULL IDENTITY(1,1),
    Procesador_Velocidad NVARCHAR(50)  NOT NULL,
    CONSTRAINT PK_Procesador PRIMARY KEY (Codigo_Procesador)
);
GO

CREATE TABLE dbo.GPU (
    Codigo_GPU  INT           NOT NULL IDENTITY(1,1),
    GPU_Memoria NVARCHAR(100)  NOT NULL,
    CONSTRAINT PK_GPU PRIMARY KEY (Codigo_GPU)
);
GO

CREATE TABLE dbo.GENERAL (
    Id_General  INT            NOT NULL IDENTITY(1,1),
    Observacion NVARCHAR(500)  NULL,
    Estado      NVARCHAR(50)   NOT NULL,
    Firma       NVARCHAR(150)  NULL,
    CONSTRAINT PK_General PRIMARY KEY (Id_General)
);
GO

CREATE TABLE dbo.MONITOR (
    Nro_Serie_Monitor NVARCHAR(50)   NOT NULL,
    Modelo            NVARCHAR(100)  NOT NULL,
    Marca             NVARCHAR(100)  NOT NULL,
    CONSTRAINT PK_Monitor PRIMARY KEY (Nro_Serie_Monitor)
);
GO

CREATE TABLE dbo.TECLADO (
    Nro_Serie_Teclado NVARCHAR(50)   NOT NULL,
    Modelo            NVARCHAR(100)  NOT NULL,
    CONSTRAINT PK_Teclado PRIMARY KEY (Nro_Serie_Teclado)
);
GO

CREATE TABLE dbo.MOUSE (
    Nro_Serie_Mouse NVARCHAR(50)   NOT NULL,
    Modelo          NVARCHAR(100)  NOT NULL,
    CONSTRAINT PK_Mouse PRIMARY KEY (Nro_Serie_Mouse)
);
GO

CREATE TABLE dbo.PARLANTE (
    Nro_Serie_Parlante NVARCHAR(50)   NOT NULL,
    Modelo             NVARCHAR(100)  NOT NULL,
    CONSTRAINT PK_Parlante PRIMARY KEY (Nro_Serie_Parlante)
);
GO

CREATE TABLE dbo.ACCESO_REMOTO (
    Codigo_Anidesk NVARCHAR(50)   NOT NULL,
    Contrasena     NVARCHAR(255)  NOT NULL,
    CONSTRAINT PK_AccesoRemoto PRIMARY KEY (Codigo_Anidesk)
);
GO

-- ============================================================
-- TABLAS CON DEPENDENCIAS DE PRIMER NIVEL
-- ============================================================

CREATE TABLE dbo.SOFTWARE (
    Id_Software          INT            NOT NULL IDENTITY(1,1),
    Id_Sistema_Operativo INT            NOT NULL,
    Ofimatica            NVARCHAR(100)  NULL,
    Antivirus            NVARCHAR(100)  NULL,
    Navegador            NVARCHAR(100)  NULL,
    Desarrollo           NVARCHAR(100)  NULL,
    Disenio              NVARCHAR(100)  NULL,
    CONSTRAINT PK_Software PRIMARY KEY (Id_Software),
    CONSTRAINT FK_Software_SistemaOperativo
        FOREIGN KEY (Id_Sistema_Operativo)
        REFERENCES dbo.SISTEMA_OPERATIVO (Id_Sistema_Operativo)
);
GO

CREATE TABLE dbo.HARWARE (
    Id_Harware        INT            NOT NULL IDENTITY(1,1),
    Id_DiscoDuro      INT            NOT NULL,
    Codigo_Procesador INT            NOT NULL,
    Codigo_GPU        INT            NOT NULL,
    Tipo              NVARCHAR(100)  NOT NULL,
    Marca_y_Modelo    NVARCHAR(150)  NOT NULL,
    Nro_de_Serie      NVARCHAR(100)  NOT NULL,
    Ram               NVARCHAR(50)   NOT NULL,
    Unidad_CD_DVDRW   NVARCHAR(50)   NULL,
    Regulador         NVARCHAR(100)  NULL,
    CONSTRAINT PK_Harware PRIMARY KEY (Id_Harware),
    CONSTRAINT FK_Harware_DiscoDuro
        FOREIGN KEY (Id_DiscoDuro)
        REFERENCES dbo.DISCO_DURO (Id_DiscoDuro),
    CONSTRAINT FK_Harware_Procesador
        FOREIGN KEY (Codigo_Procesador)
        REFERENCES dbo.PROCESADOR (Codigo_Procesador),
    CONSTRAINT FK_Harware_GPU
        FOREIGN KEY (Codigo_GPU)
        REFERENCES dbo.GPU (Codigo_GPU)
);
GO

CREATE TABLE dbo.ACCESORIO (
    Id_Accesorio       INT           NOT NULL IDENTITY(1,1),
    Nro_Serie_Monitor  NVARCHAR(50)  NOT NULL,
    Nro_Serie_Teclado  NVARCHAR(50)  NOT NULL,
    Nro_Serie_Mouse    NVARCHAR(50)  NOT NULL,
    Nro_Serie_Parlante NVARCHAR(50)  NOT NULL,
    CONSTRAINT PK_Accesorio PRIMARY KEY (Id_Accesorio),
    CONSTRAINT FK_Accesorio_Monitor
        FOREIGN KEY (Nro_Serie_Monitor)
        REFERENCES dbo.MONITOR (Nro_Serie_Monitor),
    CONSTRAINT FK_Accesorio_Teclado
        FOREIGN KEY (Nro_Serie_Teclado)
        REFERENCES dbo.TECLADO (Nro_Serie_Teclado),
    CONSTRAINT FK_Accesorio_Mouse
        FOREIGN KEY (Nro_Serie_Mouse)
        REFERENCES dbo.MOUSE (Nro_Serie_Mouse),
    CONSTRAINT FK_Accesorio_Parlante
        FOREIGN KEY (Nro_Serie_Parlante)
        REFERENCES dbo.PARLANTE (Nro_Serie_Parlante)
);
GO

-- ============================================================
-- TABLA CENTRAL
-- ============================================================

CREATE TABLE dbo.EQUIPO_INVENTARIO (
    Id_Inventario     INT           NOT NULL IDENTITY(1,1),
    Id_Accesorio      INT           NOT NULL,
    Id_Software       INT           NOT NULL,
    Id_Codigo_Almacen INT           NOT NULL,
    Codigo_Anidesk    NVARCHAR(50)  NOT NULL,
    Id_General        INT           NOT NULL,
    Id_Harware        INT           NOT NULL,
    CONSTRAINT PK_EquipoInventario PRIMARY KEY (Id_Inventario),
    CONSTRAINT FK_EqInv_Accesorio
        FOREIGN KEY (Id_Accesorio)      REFERENCES dbo.ACCESORIO (Id_Accesorio),
    CONSTRAINT FK_EqInv_Software
        FOREIGN KEY (Id_Software)       REFERENCES dbo.SOFTWARE (Id_Software),
    CONSTRAINT FK_EqInv_Almacen
        FOREIGN KEY (Id_Codigo_Almacen) REFERENCES dbo.ALMACEN (Id_Codigo_Almacen),
    CONSTRAINT FK_EqInv_AccesoRemoto
        FOREIGN KEY (Codigo_Anidesk)    REFERENCES dbo.ACCESO_REMOTO (Codigo_Anidesk),
    CONSTRAINT FK_EqInv_General
        FOREIGN KEY (Id_General)        REFERENCES dbo.GENERAL (Id_General),
    CONSTRAINT FK_EqInv_Harware
        FOREIGN KEY (Id_Harware)        REFERENCES dbo.HARWARE (Id_Harware)
);
GO

PRINT '>>> 15 tablas creadas correctamente.';
GO
